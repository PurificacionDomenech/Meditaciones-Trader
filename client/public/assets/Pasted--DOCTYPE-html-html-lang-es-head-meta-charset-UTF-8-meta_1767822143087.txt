        async function loadData() {
            try {
                const data = await window.storage.get('trader-missions-data');
                if (data) {
                    const parsed = JSON.parse(data.value);
                    completedDays = parsed.completedDays || [];
                    entriesByDate = parsed.entriesByDate || {};
                    currentDay = parsed.currentDay || 1;
                }
            } catch (e) {

            }
            updateUI();
        }

        async function completeExercise() {

            const exercise = exercises[(currentDay - 1) % exercises.length];
            
            completedDays.push(currentDay);
            entriesByDate[selectedDate] = {
                day: currentDay,
                title: exercise.title,
                challenge: exercise.challenge,
                feeling: feeling,

            currentDay++;
            document.getElementById('user-feeling').value = '';
            backToIntro();
            updateUI();
        }

                document.getElementById('calendar-view').classList.remove('hidden');
                renderCalendar();
            }
        }

        function showEntryDetail(dateStr) {
            const entry = entriesByDate[dateStr];
            if (!entry) return;
            
            const detailDiv = document.getElementById('entry-details');
            const date = new Date(dateStr + 'T00:00:00');
            const formattedDate = date.toLocaleDateString('es-ES', { 
                weekday: 'long', 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric' 
            });
            
            detailDiv.innerHTML = `
                <div class="entry-detail">
                    <h3>${entry.icon} Misi√≥n ${entry.day}: ${entry.title}</h3>
                    <p style="color: rgba(255, 215, 0, 0.7); font-size: 14px; margin-bottom: 15px;">${formattedDate}</p>

                </div>
            `;
        }

        function renderCalendar() {
            const today = new Date();
            const month = today.getMonth();
            const year = today.getFullYear();
            const monthNames = ['Enero', 'Febrero', 'Marzo', 'Abril', 'Mayo', 'Junio', 'Julio', 'Agosto', 'Septiembre', 'Octubre', 'Noviembre', 'Diciembre'];
            
            document.getElementById('calendar-month').textContent = monthNames[month] + ' ' + year;
            
            const firstDay = new Date(year, month, 1).getDay();
            const daysInMonth = new Date(year, month + 1, 0).getDate();
            
            const grid = document.getElementById('calendar-grid');
            grid.innerHTML = '';
            
            document.getElementById('entry-details').innerHTML = '';
            
            for (let i = 0; i < firstDay; i++) {
                const empty = document.createElement('div');
                grid.appendChild(empty);
            }
            
            for (let day = 1; day <= daysInMonth; day++) {
                const dateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
                const hasEntry = entriesByDate[dateStr];
                const isToday = day === today.getDate();
                
                const dayDiv = document.createElement('div');
                dayDiv.className = 'calendar-day';
                if (hasEntry) dayDiv.className += ' completed';
                if (isToday) dayDiv.className += ' today';
                dayDiv.textContent = day;
                
                if (hasEntry) {
                    const star = document.createElement('span');
                    star.className = 'star-icon';
                    star.textContent = '‚≠ê';
                    dayDiv.appendChild(star);
                    
                    dayDiv.onclick = () => showEntryDetail(dateStr);
                }
                
                grid.appendChild(dayDiv);
            }
        }
